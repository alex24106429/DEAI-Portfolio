--BikeToDrive_2_Fietsverkoop;

CREATE TABLE Fietsverkoop_Fabrikant (
	fabrikantnr INT,
	naam VARCHAR (45),
	adres VARCHAR(100),
	plaats VARCHAR(45),

	PRIMARY KEY(fabrikantnr)
);

CREATE TABLE Fietsverkoop_filiaal (
	filiaalnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	provincie VARCHAR(45),

	PRIMARY KEY(filiaalnr)
);

CREATE TABLE Fietsverkoop_klant (
	klantnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	woonplaats VARCHAR(45),
	geslacht CHAR(1),
	geboortedatum VARCHAR(10),

	PRIMARY KEY(klantnr)
);

CREATE TABLE Fietsverkoop_Monteur (
	monteurnr INT,
	naam VARCHAR(45),
	woonplaats VARCHAR(45),
	uurloon FLOAT,
	filiaal INT NOT NULL,

	PRIMARY KEY(monteurnr)
	FOREIGN KEY(filiaal) REFERENCES Fietsverkoop_Filiaal(filiaalnr)
);

CREATE TABLE Fietsverkoop_Fiets (
	fietsnr INT,
	soort VARCHAR(45),
	merk VARCHAR(45),
	type VARCHAR(45),
	standaardprijs FLOAT,
	inkoopprijs FLOAT,
	kleur VARCHAR(45),
	fabrikant INT NOT NULL,

	PRIMARY KEY(fietsnr)
	FOREIGN KEY(fabrikant) REFERENCES Fietsverkoop_fabrikant(fabrikantnr)
);

CREATE TABLE Fietsverkoop (
	fiets_verkoopnr INT,
	datum VARCHAR(10),
	aantal INT,
	verkoopprijs FLOAT,
	klant INT NOT NULL,
	fiets INT NOT NULL,
	monteur INT NOT NULL,

	PRIMARY KEY(fiets_verkoopnr)
	FOREIGN KEY(klant) REFERENCES Fietsverkoop_klant(klantnr)
	FOREIGN KEY(fiets) REFERENCES Fietsverkoop_fiets(fietsnr)
	FOREIGN KEY(monteur) REFERENCES Fietsverkoop_monteur(monteurnr)

);
